### 1. 二叉树结构
```vuejs
        if (!this.hasFollow) {
          this.trackLog(1)
          return
        }
        if (this.hasStatus) {
          this.trackLog(2)
          this.infoList = this.infoLists[0]
          return
        }
        if (this.hasNeedRecharge) {
          this.trackLog(3)
          this.infoList = this.infoLists[1]
          return
        }
        if (this.hasUpgrade) {
          this.trackLog(4)
          this.infoList = this.infoLists[2]
          return
        }
        this.trackLog(5)
        this.infoList = this.infoLists[3]
```
解决方案
```vuejs
 setGuideContent () {
   let arr = (!this.hasFollow &amp;&amp; [1, -1]) || (this.hasStatus &amp;&amp; [2, 0]) || (this.hasNeedRecharge &amp;&amp; [3, 1]) || (this.hasUpgrade &amp;&amp; [4, 2]) || [5, 3]
   this.handleData(arr[0], arr[1])
 },

     this.trackLog(type)
     if (index !== -1) {
        this.infoList = this.infoLists[index]
     }
 },
```
